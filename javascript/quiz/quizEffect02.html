// 정답 확인
quizConfirm.forEach(function(btn, num){                 // forEach로 버튼과 num을 반복함
    btn.addEventListener("click", function(){
        // 사용자 입력 정답(공백 제거, 소문자 변환)
        const userAnswer = quizInput[num].value.trim().toLowerCase();
        
        // 사용자 정답 비교
        if(userAnswer == quizInfo[num].infoAnswer){
            quizAnimaition[num].classList.add("like");      // 같으면 quizAnimaition의 선택된 요소에 class "like"를 추가
        } else {
            quizAnimaition[num].classList.add("dislike");   // 다르면 quizAnimaition의 선택된 요소에 class "dislike"를 추가
        }

        quizAnswer[num].style.display = "block";            // 정답 표시
        quizDesc[num].style.display = "block";              // 설명 표시
        quizConfirm[num].style.display = "none";            // 확인 버튼 숨김
        quizInput[num].style.display = "none";              // 입력창 숨김
    });
});
