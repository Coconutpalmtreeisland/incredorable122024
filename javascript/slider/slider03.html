// 선택자
const sliderWrap = document.querySelector(".slider__wrap");
const sliderImg = sliderWrap.querySelector(".slider__img");         // 보여지는 영역
const sliderInner = sliderWrap.querySelector(".slider__inner");     // 움직이는 영역
const slider = sliderWrap.querySelectorAll(".slider");              // 개별 이미지

// 변수 설정
let currentIndex = 0;                                               //현재 이미지
let sliderCount = slider.length;                                    // 이미지 갯수
let sliderInterval = 2000;                                          // 이미지 변경 간격 시간
let sliderWidth  = slider[0].offsetWidth;                           // 이미지 가로 값
let sliderClone = sliderInner.firstElementChild.cloneNode(true);    // 첫번째 이미지를 복사

// 복사한 이미지를 마지막에 붙여넣기
sliderInner.appendChild(sliderClone);                               // 변수에 저장된 이미지 클론을 슬라이더 내부에 추가, 슬라이더 무한 루프 설정

function sliderEffect (){                                           // 이미지 전환 효과
    currentIndex++;                                                 // 현재 표시 중인 이미지의 인덱스를 변경

    // 이미지 슬라이더의 애니메이션 효과를 설정
    sliderInner.style.transition = "all 0.6s ease-in";
    sliderInner.style.transform = "translateX(-"+ sliderWidth * currentIndex +"px)";

    // 마지막 이미지가 위치했을 때
    if(currentIndex == sliderCount){
        setTimeout(() => {
            // 애니메이션 효과 해제
            sliderInner.style.transition = "0s";
            // 이미지 슬라이더를 초기 위치로 이동
            sliderInner.style.transform = "translateX(0px)";
        }, 700)
        currentIndex = 0;
    }
}
setInterval(sliderEffect, sliderInterval);                          // 슬라이더 이펙트를 주기적으로 실행(자동으로 이미지를 전환하며 루프 동작을 수행)

// 선택자
const sliderWrap = document.querySelector(".slider__wrap");
const sliderImg = sliderWrap.querySelector(".slider__img");         // 보여지는 영역
const sliderInner = sliderWrap.querySelector(".slider__inner");     // 움직이는 영역
const slider = sliderWrap.querySelectorAll(".slider");              // 개별 이미지

// 변수 설정
let currentIndex = 0;                                               //현재 이미지
let sliderCount = slider.length;                                    // 이미지 갯수
let sliderInterval = 2000;                                          // 이미지 변경 간격 시간
let sliderWidth  = slider[0].offsetWidth;                           // 이미지 가로 값
let sliderClone = sliderInner.firstElementChild.cloneNode(true);    // 첫번째 이미지를 복사

// 복사한 이미지를 마지막에 붙여넣기
sliderInner.appendChild(sliderClone);

function sliderEffect (){
    currentIndex++;                                                 // 현재 슬라이더에서 표시 중인 이미지의 인덱스 보여줌

    gsap.to(sliderInner, {
        x: -sliderWidth * currentIndex,                             // 현재 이미지 인덱스에 따라 이동 위치를 계산
        duration: 0.6,
        ease: "power2.out"                                          // 애니메이션의 가속 및 감속 패턴을 지정
    });

    // 마지막 이미지가 위치했을 때
    if(currentIndex === sliderCount){
        setTimeout(() => {
            gsap.set(sliderInner, {x: 0});                          // 슬라이더를 처음으로 이동
            currentIndex = 0                                        // 첫 번째 이미지로 이동
        }, 700);                                                    // 0.7초 후에 초기화
    }
}
setInterval(sliderEffect, sliderInterval);                          // 슬라이더 이펙트를 주기적으로 실행(자동으로 이미지를 전환하며 루프 동작을 수행)

// 선택자
const sliderWrap = document.querySelector(".slider__wrap");
const sliderImg = sliderWrap.querySelector(".slider__img");         // 보여지는 영역
const sliderInner = sliderWrap.querySelector(".slider__inner");     // 움직이는 영역
const slider = sliderWrap.querySelectorAll(".slider");              // 개별 이미지

// 변수 설정
let currentIndex = 0;                                               // 현재 이미지
let sliderCount = slider.length;                                    // 이미지 갯수
let sliderInterval = 2000;                                          // 이미지 변경 간격 시간
let sliderWidth  = slider[0].offsetWidth;                           // 이미지 가로 값
let sliderClone = sliderInner.firstElementChild.cloneNode(true);    // 첫번째 이미지를 복사

// 복사한 이미지를 마지막에 붙여넣기
$(".slider__inner").append(sliderClone);

function sliderEffect (){
    currentIndex++;

    // 이미지 슬라이더의 애니메이션 효과를 설정
    $(".slider__inner").css({
        "transition": "all 0.6s",
        "transform": "translateX(-"+ sliderWidth * currentIndex +"px)"  // 현재 이미지 인덱스에 따라 이동 위치를 계산
    });

    // 마지막 이미지가 위치했을 때
    if(currentIndex === sliderCount){
        setTimeout(() => {
            // 애니메이션 효과 해제
            $(".slider__inner").css({
                "transition": "0s",
                "transform": "translateX(0px)"                      // 슬라이더를 처음으로 이동
            });
        }, 700);                                                    // 0.7초 후에 초기화
        currentIndex = 0                                            // 첫 번째 이미지로 이동
    }
}
setInterval(sliderEffect, sliderInterval);                          // 슬라이더 이펙트를 주기적으로 실행(자동으로 이미지를 전환하며 루프 동작을 수행)
